    <script type="text/babel">

        const { useState, useEffect, useRef } = React;

 

        // --- Configuration & Data ---

        const A2L_PATHWAY = {

            { range: [13, 14], summary: "All 72 \"Learn Its\" embedded; division using multiples and remainders." },

            { range: [15, 16], summary: "Dividing 3-4 digits; decimal calculation; long multiplication mastery." },

            { range: [17, 18], summary: "Fractions, percentages, decimals; negative numbers; advanced column methods." },

            { range: [19, 20], summary: "Rapid calculation; complex multi-step problem solving; KS3 preparation." }

        ];

 

        const TARGET_A2L = { "Nursery": 1, "Reception": 4, "1": 5, "2": 6, "3": 7, "4": 8, "5": 9, "6": 10 };

        const TARGET_CLIC = { "Nursery": 1, "Reception": 2, "1": 4, "2": 8, "3": 10, "4": 14, "5": 17, "6": 20 };

 

        const SUBJECT_LISTS = {

            KS1: [ {id:'re', name:'RE'}, {id:'sci', name:'Science'}, {id:'hist', name:'History'}, {id:'geo', name:'Geography'}, {id:'art', name:'Art'}, {id:'comp', name:'Computing'}, {id:'mus', name:'Music'}, {id:'pe', name:'PE'}, {id:'pshe', name:'PSHE'} ],

            LKS2: [ {id:'re', name:'RE'}, {id:'sci', name:'Science'}, {id:'hist', name:'History'}, {id:'geo', name:'Geography'}, {id:'art', name:'Art'}, {id:'comp', name:'Computing'}, {id:'mus', name:'Music'}, {id:'pe', name:'PE'}, {id:'pshe', name:'PSHE'} ],

            UKS2: [ {id:'re', name:'RE'}, {id:'sci', name:'Science'}, {id:'hist', name:'History'}, {id:'geo', name:'Geography'}, {id:'art', name:'Art'}, {id:'dt', name:'DT'}, {id:'comp', name:'Computing'}, {id:'mus', name:'Music'}, {id:'pe', name:'PE'}, {id:'pshe', name:'PSHE'} ]

        };

 

        const Icon = ({ name, size = 20, className = "" }) => {

            return <div dangerouslySetInnerHTML={{ __html: lucide.icons[name]?.toSvg({ size, class: className }) }} style={{ display: 'inline-flex' }} />;

        };

 

        const GradeBadge = ({ grade }) => {

            const styles = {

                'Greater Depth': 'bg-emerald-100 text-emerald-800 border-emerald-200',

                'Expected': 'bg-emerald-50 text-emerald-900 border-emerald-100',

                'Working Towards': 'bg-amber-50 text-amber-700 border-amber-100'

            };

            return <span className={`inline-block px-3 py-1 rounded-full text-[9px] font-black border uppercase tracking-wider ${styles[grade] || ''}`}>{grade}</span>;

        };

 

        function App() {

            // --- State ---

            const [profile, setProfile] = useState({

                name: '', yearGroup: '6', attendance: 100, ageYears: 11, ageMonths: 0,

                readingAge: '', clicScore: 0, satsReading: '', satsGps: '', satsMaths: '',

                nferReading: '', nferMaths: ''

            });

            const [judgments, setJudgments] = useState({ reading: 'Expected', writing: 'Expected', maths: 'Expected', science: 'Expected' });

            const [a2lLevel, setA2lLevel] = useState(10);

            const [rubricState, setRubricState] = useState({});

            const [aiCommentary, setAiCommentary] = useState('');

            const [isGenerating, setIsGenerating] = useState(false);

            const [errorMsg, setErrorMsg] = useState('');

            const [showSettings, setShowSettings] = useState(false);

            const [userApiKey, setUserApiKey] = useState('');

 

            const reportRef = useRef(null);

 

            // --- Effects ---

            useEffect(() => {

                const savedKey = localStorage.getItem('tp_report_genius_key');

                if (savedKey) setUserApiKey(savedKey);

            }, []);

 

            // --- Handlers ---

            const handleProfileChange = (e) => setProfile(prev => ({ ...prev, [e.target.id]: e.target.value }));

           

            const handleYearChange = (e) => {

                const yr = e.target.value;

                setProfile(prev => ({ ...prev, yearGroup: yr }));

                setA2lLevel(TARGET_A2L[yr] || 10);

                setRubricState({});

            };

 

            const getActiveSubjects = () => {

                const yr = profile.yearGroup;

                if (['Nursery', 'Reception', '1', '2'].includes(yr)) return SUBJECT_LISTS.KS1;

                if (['3', '4'].includes(yr)) return SUBJECT_LISTS.LKS2;

                return SUBJECT_LISTS.UKS2;

            };

 

            const generateAI = async () => {

                if (!profile.name) {

                    setErrorMsg("Please enter a pupil name before generating.");

                    return;

                }

                if (!userApiKey) { setShowSettings(true); return; }

 

                setIsGenerating(true);

                setErrorMsg('');

                setAiCommentary('');

 

                const activeSubs = getActiveSubjects();

                const rubricSummary = activeSubs

                    .filter(s => rubricState[s.id] !== undefined)

                    .map(s => `${s.name}: ${['Emerging', 'Developing', 'Secure'][rubricState[s.id]]}`)

                    .join(', ');

 

                // COMBINED PROMPT FOR MAXIMUM COMPATIBILITY

                const promptText = `

                    ACT AS: An expert Primary School Teacher at Taddington & Priestcliffe C of E Federation.

                    CONTEXT: We are a school focused on "FLOURISHING TOGETHER". Our core values are Respect, Resilience, Motivate, Aspire, Together, Imagine, and Forgiveness.

                    TONE: Professional, warm, and encouraging. Use British English, third person.

                   

                    PUPIL DATA:

                    - Pupil Name: ${profile.name}

                    - Year Group: ${profile.yearGroup}

                    - Attitude to Learning (A2L) Level: ${a2lLevel} (Phase: ${A2L_PATHWAY[a2lLevel].phase})

                    - Core Subject Judgments: Reading: ${judgments.reading}, Writing: ${judgments.writing}, Mathematics: ${judgments.maths}, Science: ${judgments.science}

                    - Scores: Reading Age ${profile.readingAge}, CLIC Score ${profile.clicScore}

                    - Foundation Subject Highlights: ${rubricSummary}

 

                    TASK: Write a holistic summative report commentary (approx 180-220 words). Ensure the commentary reflects the school's vision and values while synthesising the child's academic and social progress.

                `;

 

                try {

                    // Using v1beta as it handles direct text prompts most reliably in web environments

                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${userApiKey}`, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify({ contents: [{ parts: [{ text: promptText }] }] })

                    });

                   

                    const result = await response.json();

                    if (!response.ok) throw new Error(result.error?.message || 'Connection Error. Please check your API key.');

                    setAiCommentary(result.candidates[0].content.parts[0].text);

                } catch (err) {

                    setErrorMsg(`AI ERROR: ${err.message}. If this error persists, ensure your Gemini API Key is valid.`);

                } finally {

                    setIsGenerating(false);

                }

            };

 

            const downloadPDF = async () => {

                if (!reportRef.current) return;

                const canvas = await html2canvas(reportRef.current, { scale: 2, useCORS: true, backgroundColor: '#ffffff' });

                const imgData = canvas.toDataURL('image/png');

                const pdfWidth = 210;

                const pdfHeight = (canvas.height * pdfWidth) / canvas.width;

                const pdf = new jspdf.jsPDF({ orientation: 'p', unit: 'mm', format: [pdfWidth, pdfHeight] });

                pdf.addImage(imgData, 'PNG', 0, 0, pdfWidth, pdfHeight);

                pdf.save(`${profile.name.replace(/\s+/g, '_') || 'Report'}.pdf`);

            };

 

            const currentClicData = CLIC_PROGRESSION.find(p => profile.clicScore >= p.range[0] && profile.clicScore <= p.range[1]);

 

            return (

                <div className="max-w-7xl mx-auto p-4 md:p-8">

                    {/* Settings Modal */}

                    {showSettings && (

                        <div className="fixed inset-0 bg-slate-900/70 backdrop-blur-sm z-[100] flex items-center justify-center p-4">

                            <div className="bg-white rounded-[2rem] p-8 max-w-md w-full shadow-2xl">

                                <h2 className="text-2xl font-bold mb-2 flex items-center gap-2 text-emerald-900"><Icon name="key" /> API Configuration</h2>

                                <p className="text-slate-500 text-sm mb-6 uppercase tracking-widest font-bold">Google AI Studio Key Required</p>

                                <input

                                    type="password"

                                    value={userApiKey}

                                    onChange={(e) => setUserApiKey(e.target.value)}

                                    placeholder="Paste Gemini API Key..."

                                    className="w-full border-2 border-slate-100 rounded-2xl p-4 mb-4 focus:border-emerald-500 outline-none transition-all"

                                />

                                <button

                                    onClick={() => { localStorage.setItem('tp_report_genius_key', userApiKey); setShowSettings(false); }}

                                    className="w-full bg-emerald-600 text-white font-black py-4 rounded-2xl hover:bg-emerald-700 transition shadow-lg"

                                >

                                    SAVE & CONNECT

                                </button>

                                <button onClick={() => setShowSettings(false)} className="w-full mt-4 text-slate-400 font-bold uppercase text-[10px] tracking-widest">Close</button>

                            </div>

                        </div>

                    )}

 

                    {/* App Header */}

                    <header className="mb-8 flex flex-col xl:flex-row justify-between items-start xl:items-center gap-6 no-print">

                        <div className="flex items-center gap-4">

                            <div className="bg-emerald-900 text-white p-3 rounded-2xl shadow-xl">

                                <Icon name="graduation-cap" size={40} />

                            </div>

                            <div>

                                <h1 className="text-4xl font-black text-emerald-900 tracking-tighter">Report Genius</h1>

                                <p className="text-slate-500 font-bold text-sm uppercase tracking-widest">Taddington & Priestcliffe Federation</p>

                            </div>

                        </div>

                        <div className="flex flex-wrap gap-2 w-full xl:w-auto">

                            <button onClick={() => setShowSettings(true)} className="flex-1 bg-white border-2 border-slate-100 text-slate-600 px-6 py-3 rounded-2xl font-bold flex items-center justify-center gap-2 hover:bg-slate-50 transition shadow-sm"><Icon name="settings" size={18}/> API Config</button>

                            <button onClick={downloadPDF} className="flex-1 bg-emerald-700 text-white px-6 py-3 rounded-2xl font-bold flex items-center justify-center gap-2 hover:bg-emerald-800 transition shadow-lg"><Icon name="file-text" size={18}/> Borderless PDF</button>

                            <button onClick={() => window.print()} className="flex-1 bg-slate-800 text-white px-6 py-3 rounded-2xl font-bold flex items-center justify-center gap-2 hover:bg-slate-900 transition shadow-md"><Icon name="printer" size={18}/> Print</button>

                        </div>

                    </header>

 

                    <div className="grid grid-cols-1 lg:grid-cols-12 gap-8">

                        {/* LEFT: INPUTS */}

                        <aside className="lg:col-span-4 space-y-6 no-print">

                            {/* Profile Section */}

                            <div className="bg-white p-6 rounded-3xl shadow-sm border border-slate-200">

                                <div className="space-y-4">

                                    <input type="text" id="name" value={profile.name} onChange={handleProfileChange} placeholder="Pupil Full Name" className="w-full border-2 border-slate-50 bg-slate-50/50 rounded-xl p-3 text-sm focus:bg-white focus:border-emerald-500 outline-none transition-all" />

                                    <div className="grid grid-cols-2 gap-4">

                                        <select value={profile.yearGroup} onChange={handleYearChange} className="border-2 border-slate-50 bg-slate-50/50 rounded-xl p-3 text-sm outline-none appearance-none">

                                            {['Nursery', 'Reception', '1', '2', '3', '4', '5', '6'].map(y => <option key={y} value={y}>Year {y}</option>)}

                                        </select>

                                            <input type="number" id="attendance" value={profile.attendance} onChange={handleProfileChange} placeholder="Att %" className="w-full border-2 border-slate-50 bg-slate-50/50 rounded-xl p-3 text-sm" />

                                            <span className="absolute right-3 top-3 text-[10px] font-bold text-slate-300">%</span>

                                        </div>

                                    </div>

                                </div>

                            </div>

 

                            {/* Scores & Tests */}

                            <div className="bg-white p-6 rounded-3xl shadow-sm border border-slate-200">

                                <h2 className="text-[11px] font-black text-slate-400 uppercase mb-4 tracking-widest flex items-center gap-2"><Icon name="calculator" size={14}/> Academic Data</h2>

                                <div className="grid grid-cols-2 gap-4 mb-4">

                                    <div className="space-y-1">

                                        <label className="text-[9px] font-black text-slate-400 uppercase ml-1">Reading Age</label>

                                        <input type="text" id="readingAge" value={profile.readingAge} onChange={handleProfileChange} placeholder="e.g. 11.08" className="w-full border-2 border-slate-50 bg-slate-50/50 rounded-xl p-3 text-xs" />

                                    </div>

                                    <div className="space-y-1">

                                        <label className="text-[9px] font-black text-slate-400 uppercase ml-1">CLIC Score</label>

                                        <input type="number" id="clicScore" value={profile.clicScore} onChange={handleProfileChange} className="w-full border-2 border-slate-50 bg-slate-50/50 rounded-xl p-3 text-xs" />

                                    </div>

                                </div>

                                {profile.yearGroup === '6' ? (

                                    <div className="bg-emerald-50/50 p-4 rounded-2xl border-2 border-emerald-50 space-y-3">

                                        <p className="text-[10px] font-black text-emerald-800 uppercase tracking-widest">SATs Standardised</p>

                                        <div className="grid grid-cols-3 gap-2">

                                            <input type="number" id="satsReading" value={profile.satsReading} onChange={handleProfileChange} placeholder="R" className="p-2 text-xs rounded-lg border-none shadow-sm" />

                                            <input type="number" id="satsGps" value={profile.satsGps} onChange={handleProfileChange} placeholder="GPS" className="p-2 text-xs rounded-lg border-none shadow-sm" />

                                            <input type="number" id="satsMaths" value={profile.satsMaths} onChange={handleProfileChange} placeholder="M" className="p-2 text-xs rounded-lg border-none shadow-sm" />

                                        </div>

                                    </div>

                                ) : ['2','3','4','5'].includes(profile.yearGroup) && (

                                    <div className="bg-slate-50 p-4 rounded-2xl border-2 border-slate-100 space-y-3">

                                        <p className="text-[10px] font-black text-slate-500 uppercase tracking-widest">NFER Scores</p>

                                        <div className="grid grid-cols-2 gap-2">

                                            <input type="number" id="nferReading" value={profile.nferReading} onChange={handleProfileChange} placeholder="Reading" className="p-2 text-xs rounded-lg border-none shadow-sm" />

                                            <input type="number" id="nferMaths" value={profile.nferMaths} onChange={handleProfileChange} placeholder="Maths" className="p-2 text-xs rounded-lg border-none shadow-sm" />

                                        </div>

                                    </div>

                                )}

                            </div>

 

                            {/* Teacher Judgments */}

                            <div className="bg-white p-6 rounded-3xl shadow-sm border border-slate-200">

                                <h2 className="text-[11px] font-black text-slate-400 uppercase mb-4 tracking-widest flex items-center gap-2"><Icon name="award" size={14}/> Core Judgments</h2>

                                <div className="space-y-3">

                                    {['reading', 'writing', 'maths', 'science'].map(sub => (

                                        <div key={sub} className="flex items-center justify-between gap-4">

                                            <label className="text-[10px] font-black text-slate-600 uppercase">{sub}</label>

                                            <select value={judgments[sub]} onChange={(e) => setJudgments(p => ({...p, [sub]: e.target.value}))} className="bg-slate-50 border-2 border-slate-50 rounded-xl p-2 text-[10px] font-bold outline-none">

                                                <option>Working Towards</option><option>Expected</option><option>Greater Depth</option>

                                            </select>

                                        </div>

                                    ))}

                                </div>

                            </div>

 

                            {/* Rubrics */}

                            <div className="bg-white p-6 rounded-3xl shadow-sm border border-slate-200">

                                <h2 className="text-[11px] font-black text-slate-400 uppercase mb-4 tracking-widest">Foundation Rubric</h2>

                                <div className="space-y-4 max-h-[350px] overflow-y-auto pr-2 custom-scrollbar">

                                    {getActiveSubjects().map(sub => (

                                        <div key={sub.id} className="p-3 border-b border-slate-50 last:border-0">

                                            <label className="block text-[10px] font-bold text-slate-700 mb-2">{sub.name}</label>

                                            <div className="flex gap-1">

                                                {['Emerging', 'Developing', 'Secure'].map((lvl, idx) => (

                                                    <button

                                                        key={lvl}

                                                        onClick={() => setRubricState(prev => ({...prev, [sub.id]: idx}))}

                                                        className={`flex-1 text-[8px] py-2 rounded-lg font-black transition-all ${rubricState[sub.id] === idx ? 'bg-emerald-600 text-white shadow-md' : 'bg-slate-50 text-slate-400'}`}

                                                    >

                                                        {lvl.toUpperCase()}

                                                    </button>

                                                ))}

                                            </div>

                                        </div>

                                    ))}

                                </div>

                            </div>

 

                            <button disabled={isGenerating} onClick={generateAI} className="w-full bg-emerald-700 text-white font-black py-5 rounded-[1.5rem] shadow-xl hover:bg-emerald-800 transition-all flex items-center justify-center gap-3 disabled:opacity-50 group">

                                <Icon name="sparkles" size={20} className={isGenerating ? "animate-spin" : "group-hover:animate-pulse"} />

                                {isGenerating ? "SYNTESISING..." : "GENERATE AI COMMENTARY"}

                            </button>

                        </aside>

 

                        {/* RIGHT: PREVIEW */}

                        <main className="lg:col-span-8">

                            <div ref={reportRef} id="reportCard" className="bg-white rounded-[2.5rem] shadow-2xl border border-slate-200 overflow-hidden flex flex-col min-h-full">

                                {/* Report Header */}

                                <div className="bg-emerald-900 text-white p-12 flex flex-col md:flex-row justify-between items-start">

                                    <div className="space-y-1">

                                        <h2 className="text-4xl serif-heading italic font-bold">Academic Report</h2>

                                        <p className="text-emerald-300 text-xs font-black uppercase tracking-[0.2em] border-l-4 border-emerald-500 pl-3">Taddington & Priestcliffe Federation</p>

                                    </div>

                                    <div className="text-left md:text-right mt-6 md:mt-0 uppercase">

                                        <p className="font-black text-3xl tracking-tighter">{profile.name || 'Student Name'}</p>

                                        <p className="text-emerald-400 text-sm font-bold tracking-widest">Year {profile.yearGroup} • 2025/26</p>

                                    </div>

                                </div>

 

                                <div className="p-10 space-y-10 flex-grow">

                                    {/* Stats Tiles */}

                                    <div className="grid grid-cols-2 md:grid-cols-4 gap-6">

                                        <div className="p-5 bg-slate-50 rounded-3xl border border-slate-100 text-center">

                                            <p className="text-[10px] text-slate-400 uppercase font-black mb-1">A2L Grade</p>

                                            <p className="text-3xl font-black text-emerald-900">L{a2lLevel}</p>

                                        </div>
